import java.util.*;

class Solution {
    public int nearestExit(char[][] maze, int[] entrance) {
        int m = maze.length, n = maze[0].length;
        int sr = entrance[0], sc = entrance[1];

        // 4-directional moves
        int[][] dirs = {{1,0},{-1,0},{0,1},{0,-1}};

        // BFS queue: {row, col, dist}
        ArrayDeque<int[]> q = new ArrayDeque<>();
        q.offer(new int[]{sr, sc, 0});
        // Mark entrance as visited so we don't step back onto it
        maze[sr][sc] = '+';

        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int r = cur[0], c = cur[1], d = cur[2];

            for (int[] dir : dirs) {
                int nr = r + dir[0], nc = c + dir[1];
                if (nr < 0 || nr >= m || nc < 0 || nc >= n) continue;   // out of bounds
                if (maze[nr][nc] != '.') continue;                       // wall or visited

                // If it's on the border and not the entrance, it's the nearest exit
                if (nr == 0 || nr == m - 1 || nc == 0 || nc == n - 1) {
                    return d + 1;
                }

                maze[nr][nc] = '+';                    // mark visited
                q.offer(new int[]{nr, nc, d + 1});
            }
        }
        return -1; // no exit reachable
    }
}
