fun numTilings(n: Int): Int {
    val MOD = 1_000_000_007

    // dp[i]: 2 x i 크기 보드를 덮는 경우의 수
    if (n == 0) return 1
    if (n == 1) return 1
    if (n == 2) return 2

    val dp = IntArray(n + 1)
    dp[0] = 1 // 빈 보드
    dp[1] = 1 // 도미노 하나
    dp[2] = 2 // 도미노2개 또는 트로미노1개

    for (i in 3..n) {
        // 점화식: dp[i] = 2*dp[i-1] + dp[i-3]
        dp[i] = (((2L * dp[i - 1]) % MOD + dp[i - 3]) % MOD).toInt()
    }

    return dp[n]
}
