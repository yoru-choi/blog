875. Koko Eating Bananas

```kotlin
fun minEatingSpeed(piles: IntArray, h: Int): Int {
    var left = 1
    var right = piles.max()!!
    var answer = right

    while (left <= right) {
        val mid = left + (right - left) / 2
        if (canEatAll(piles, h, mid)) {
            answer = mid
            right = mid - 1
        } else {
            left = mid + 1
        }
    }

    return answer
}

fun canEatAll(piles: IntArray, h: Int, k: Int): Boolean {
    var time = 0L
    for (pile in piles) {
        // 올림: 한 더미에 대해 ceil(pile / k)
        time += (pile + k - 1) / k
    }
    return time <= h
}
```
