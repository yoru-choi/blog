class Solution {
    fun reverseList(head: ListNode?): ListNode? {
        var prev: ListNode? = null
        var curr = head

        while (curr != null) {
                     2          3
            val nextTemp = curr.next  // 현재 노드의 다음 노드를 저장
              null         null
            curr.next = prev          // 방향을 뒤집음
              1      1
            prev = curr               // prev 이동
              2      2
            curr = nextTemp           // curr 이동
        }

        return prev  // prev가 새로운 헤드가 됨
    }
}
