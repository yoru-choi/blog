class Solution {
    fun closeStrings(word1: String, word2: String): Boolean {
        // Step 1: 길이 다르면 false
        if (word1.length != word2.length) return false

        // Step 2: 수동 빈도수 맵 만들기
        val map1 = mutableMapOf<Char, Int>()
        for (ch in word1) {
            map1[ch] = map1.getOrDefault(ch, 0) + 1
        }

        val map2 = mutableMapOf<Char, Int>()
        for (ch in word2) {
            map2[ch] = map2.getOrDefault(ch, 0) + 1
        }

        // Step 3: 문자 집합(Set)이 같은지 비교
        if (map1.keys != map2.keys) return false

        // Step 4: 빈도 수 정렬 후 비교
        val freqList1 = map1.values.sorted()
        val freqList2 = map2.values.sorted()

        return freqList1 == freqList2
    }
}

class Solution {
    fun closeStrings(word1: String, word2: String): Boolean {
        // Step 1: 길이가 다르면 바로 false
        if (word1.length != word2.length) return false

        // Step 2: 문자 빈도수 맵 만들기
        val freq1 = word1.groupingBy { it }.eachCount()
        val freq2 = word2.groupingBy { it }.eachCount()

        // Step 3: 문자 집합(Set)이 같은지 비교
        if (freq1.keys != freq2.keys) return false

        // Step 4: 등장 빈도값들만 정렬해서 비교
        val freqList1 = freq1.values.sorted()
        val freqList2 = freq2.values.sorted()

        return freqList1 == freqList2
    }
}
