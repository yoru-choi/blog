class Solution {
    fun equalPairs(grid: Array<IntArray>): Int {
        val n = grid.size

        val rowMap = mutableMapOf<String, Int>()
        for (row in grid) {
            val key = row.joinToString(",")
            rowMap[key] = rowMap.getOrDefault(key, 0) + 1
        }

        var count = 0
        for (col in 0 until n) {
            val column = IntArray(n) { row -> grid[row][col] }
            val key = column.joinToString(",")
            count += rowMap.getOrDefault(key, 0)
        }

        return count
    }
}
