import java.util.*;

class Solution {
    public int findCircleNum(int[][] isConnected) {
        int n = isConnected.length;
        boolean[] seen = new boolean[n];
        int provinces = 0;

        for (int i = 0; i < n; i++) {
            if (!seen[i]) {
                provinces++;
                dfs(i, isConnected, seen, n);
            }
        }
        return provinces;
    }

    private void dfs(int u, int[][] g, boolean[] seen, int n) {
        seen[u] = true;
        for (int v = 0; v < n; v++) {
            if (g[u][v] == 1 && !seen[v]) {
                dfs(v, g, seen, n);
            }
        }
    }
}
