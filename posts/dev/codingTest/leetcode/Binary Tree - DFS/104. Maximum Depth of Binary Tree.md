```kotlin
class Solution {
    fun maxDepth(root: TreeNode?): Int {
        // 1. Base case: null이면 0 리턴 (리프 노드의 다음 단계)
        if (root == null) return 0

        // 2. 재귀적으로 왼쪽, 오른쪽의 최대 깊이를 구함
        val leftDepth 0 = maxDepth(root.left)       null
        val rightDepth 1 = maxDepth(root.right)    2

        // 3. 둘 중 더 깊은 값에 +1 (현재 노드 포함)
        return maxOf(leftDepth, rightDepth) + 1
    }
}
```
