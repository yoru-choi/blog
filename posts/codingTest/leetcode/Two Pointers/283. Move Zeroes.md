```kotlin
class Solution {
    fun moveZeroes(nums: IntArray): Unit {
        var insertPos = 0
        for (i in nums.indices) {
            if (nums[i] != 0) {
                nums[insertPos] = nums[i]
                insertPos++
            }
        }
        for (i in insertPos until nums.size) {
            nums[i] = 0
        }
    }
}
```
