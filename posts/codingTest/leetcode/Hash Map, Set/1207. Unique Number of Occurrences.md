```kotlin
class Solution {
    fun uniqueOccurrences(arr: IntArray): Boolean {
        val countMap = mutableMapOf<Int, Int>()

        for (num in arr) {
            countMap[num] = countMap.getOrDefault(num, 0) + 1
        }

        val occurrenceList = countMap.values.toList()
        val occurrenceSet = occurrenceList.toSet()

        return occurrenceList.size == occurrenceSet.size
    }
}
```

```kotlin
class Solution {
    fun uniqueOccurrences(arr: IntArray): Boolean {
        // Step 1: 각 숫자의 등장 횟수 세기
        val countMap = arr.groupingBy { it }.eachCount()  // Map<Int, Int>

        // Step 2: 등장 횟수들만 따로 모으기
        val counts = countMap.values

        // Step 3: 등장 횟수 중복 확인
        return counts.size == counts.toSet().size
    }
}
```
